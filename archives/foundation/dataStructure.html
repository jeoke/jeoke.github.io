开链法

即：存储的时候，行存键值，列存数据，计算的hash值不变

  const H = 37;

    tempSet.add(this.dataStore[i]);

function selectSort (arr) {
  var min,temp;

            min = outer;
            for(var inner = outer+1;inner<=len-1;++inner){

function insertionSort (arr) {
  var temp,inner;
  for(var outer=1;outer <= arr.length-1;++outer){
    temp = arr[outer];
    inner=outer;
    while (inner > 0 && (arr[inner-1])>=temp) {
      arr[inner] = arr[inner-1];
      --inner;
    }
    arr[inner] = temp;
  }
 }

这个算法的原理是从后往前，将比较大的依次往前插入

function mergeSort (arr) {
  if(arr.length < 2){
    return;
  }
    var step = 1;
    var left,right;
    while(step < arr.length){
       left = 0;
         right = step;
         while(right+step <= arr.length){//判断是否能对半分割
               mergeArrays(arr,left,left+step,right,right+step);
               left = right+step;
               right = left+step;
         }
         if(right < arr.length){        //不能对半分，就不均分
             mergeArrays(arr,left,left+step,right,arr.length);
         }
        step*=2;
    }
 }

 function mergeArrays (arr,startLeft,stopLeft,startRight,stopRight) {
         var leftArr = new Array(stopLeft-startLeft+1);
         var rightArr = new Array(stopRight-startRight+1);
         var k = startRight;
         for(var i = 0;i < rightArr.length-1;++i){  // 传入数值
             rightArr[i] = arr[k];
             ++k; 
         }
         var h = startLeft;
         for(var i = 0;i < leftArr.length-1;++i){
             leftArr[i] = arr[h];
             ++h;
         }
         var m=0,
             n=0;
         leftArr[leftArr.length-1] = Infinity;
         rightArr[rightArr.length-1] = Infinity;
         for (var j = startLeft;j < stopRight;j++) {     //比较数组,因为左边数组长，当比较完成时，结果直接呈现在原数组上
          if(leftArr[m] < rightArr[n]){
            arr[j] = leftArr[m];
              m++;
          }else{
             arr[j] =rightArr[n];
             n++;
          }
         }
 }

function quickSort (arr) {
  if(arr.length == 0){
         return [];
  }
  var baseNum = arr[0];
  var lesser = [];
  var greater = [];
  for(var i=1;i < arr.length; ++i){
    if(arr[i] < baseNum){
      lesser.push(arr[i]);
    }else{
      greater.push(arr[i]);
    }
  }
  return quickSort(lesser).concat(baseNum,quickSort(greater));
}

function binSearch (arr,data) {
  var upperBound = arr.length-1;
  var lowerBound = 0;
  while (lowerBound <= upperBound) {
    var mid = Math.floor((upperBound + lowerBound)/2);
    if(arr[mid] < data){
      lowerBound = mid + 1;
    }else if(arr[mid] > data){
      upperBound = mid-1;
    }else{
      return mid;
    }
  }

  return -1;
}

时间复杂度为O(n),空间复杂度为O(1)
